Algoritmo Familias

Inicio

Var
	string nom_r, confirmacion, salida
	int cant_g, cant_p, total_a, sum_p = 0, sum_g = 0
	float cant_pp, cant_pg

	Hacer

		Hacer
			Escribir "Ingresa tu nombre"
			Leer nom_r

			Escribir "Cuantos Gatos tienes"
			Leer cant_g
			sum_g += cant_g

			Escribir "Cuantos Perros tienes"
			Leer cant_p
			sum_p += cant_p
		
			Escribir nom_r, cant_g, cant_p
			Escribir "Los datos son correctos? si/no"
			Leer confirmacion
		Mientras (confirmacion == "no")

		Escribir "Quiere agregar a otra persona? si/no"
		Leer salida
	Mientras (salida == "si")

	total_a = sum_g + sum_p	
	cant_pp = (sum_p / total_a) * 100
	cant_pg = (sum_g / total_a) * 100

	Escribir "El porcentaje de perros y gatos es de :", cant_pp, cant_pg

Fin
