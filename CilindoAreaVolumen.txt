Algoritmo CilindroAreaVolumen
Inicio

Var
	float Radio, Altura, Area, Volumen
Const
	Pi = 3.141592

	Escribir "Introduce la altura del circulo"
	Leer Altura

	Escribir "Introduce el Radio del circulo"
	Leer Radio

	Volumen = pi * (Radio^2) * Altura
	Area = 2 * pi * Radio * Altura

	Escribir "Volumen", Volumen
	Escribir "Area", Area

Fin
