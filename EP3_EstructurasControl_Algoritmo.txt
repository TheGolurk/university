algoritmo CensoDeMascotas

Inicio

	struct Familia{
		string NombreCuidador
		int NumeroPerros
		int NumeroGatos
	}

	VAR
		string salida, confirmacion
		Familia Familias[]
		int indice = 0, numeroGatos = 0, numeroPerros = 0, totalAnimales = 0
		float porcentajeGatos, porcentajePerros

	Hacer 

		Hacer 
	
			Escribir "Ingrese su nombre"
			leer Familias[indice].NombreCuidador

			Escribir "Cuantos perros tienes?"
			Leer Familias[indice].NumeroPerros

			Escribir "Cuantos gatos tienes?"
			Leer Familias[indice].NumeroGatos

			Escribir Familias[indice].NombreCuidador, Familias[indice].NumeroPerros, Familias[indice].NumeroGatos
			Escribir "Los Datos son correctos? si/no"
			Leer confirmacion

			Si (confirmacion == "si" || confirmacion == "SI") 
				indice++
			finSi

		Hasta (confirmacion == "si" || confirmacion == "SI") 
	


		Escribir "Continuar con otra persona?"
		Leer salida

	Hasta(salida == "si" || salida == "SI")

	Para (i = 0 ; i < longitud(Familias) ; i++) 

		numeroPerros += Familias[i].NumeroPerros
		numeroGatos += Familias[i].NumeroGatos
		totalAnimales += (numeroGatos + numeroPerros)

	FinPara

	porcentajeGatos = (numeroGatos / totalAnimales) * 100
	porcentajePerros = (numeroPerros / totalAnimales) * 100

	Escribir "El porcentaje de Gatos es de: ", porcentajeGatos
	Escribir "El porcentaje de Perros es de: ", porcentajePerros

Fin
